const arr1 = document.getElementById('1'); const arr2 = document.getElementById('2'); const btn1 = document.getElementById('add-1'); const btn2 = document.getElementById('add-2'); const firstArray = []; const secondArray = []; btn1.addEventListener('click', () => { const splittedString = arr1.value.split(','); splittedString.map(item => { if(isNaN(item)) {document.getElementById('error').classList.remove("d-none");} else { firstArray.push(Number(item)); document.getElementById('disp-1').innerHTML= "["+arr1.value+"]"; document.getElementById('error').classList.add("d-none"); } }) }) btn2.addEventListener('click', () => { const splittedString = arr2.value.split(','); splittedString.map(item => { if(isNaN(item)) {document.getElementById('error').classList.remove("d-none");} else { secondArray.push(Number(item)); document.getElementById('disp-2').innerHTML= "["+arr2.value+"]"; document.getElementById('error').classList.add("d-none"); } }) }) document.getElementById('merge').addEventListener('click', () => { const mergedArray = [...firstArray, ...secondArray] document.getElementById('result').innerHTML= "["+mergedArray+"]" })
